Task: Run a Basic Dart Program

Copy and execute the following code using dart hello.dart:


void main() {
  int age = 20;
  double height = 5.3;
  String name = 'PREETHI VINNAKOTA';
  bool isStudent = true;

  print('Name: $name');
  print('Age: $age');
  print('Height: $height');
  print('Is a student: $isStudent');

  if (age > 18) {
    print('$name is an adult.');
  } else {
    print('$name is a minor.');
  }

  print('Counting from 1 to 5:');
  for (int i = 1; i <= 5; i++) {
    print(i);
  }

  greet(name);
}

void greet(String personName) {
  print('Hello, $personName! Welcome to Dart.');
}
