Week 4: Implementing Rows, Columns & Stack Widgets

Description: Implementing various layout structures using Rows, Column and Stack widgets in one application.

Task 1: Row Layout Example


import 'package:flutter/material.dart';

// Entry point of the app
void main() {
  runApp(MyApp());
}

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Row Layout Example',
      theme: ThemeData(primarySwatch: Colors.blue),
      home: RowExample(),
    );
  }
}

// Row Layout Example
class RowExample extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text("Row Layout Example"), centerTitle: true),
      body: Center(
        child: Row(
          mainAxisAlignment: MainAxisAlignment.spaceAround,
          children: [
            Icon(Icons.home, size: 60, color: Colors.blue),
            Icon(Icons.star, size: 60, color: Colors.orange),
            Icon(Icons.settings, size: 60, color: Colors.green),
          ],
        ),
      ),
    );
  }
}

Task 3: Stack Layout Example


import 'package:flutter/material.dart';

// Entry point of the app
void main() {
  runApp(MyApp());
}

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Stack Layout Example',
      theme: ThemeData(primarySwatch: Colors.blue),
      home: StackExample(),
    );
  }
}

// Stack Layout Example
class StackExample extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text("Stack Layout Example"), centerTitle: true),
      body: Center(
        child: Stack(
          alignment: Alignment.center,
          children: [
            Container(width: 200, height: 200, color: Colors.blue),
            Container(width: 150, height: 150, color: Colors.red),
            Container(width: 100, height: 100, color: Colors.green),
            Text(
              "Stacked",
              style: TextStyle(color: Colors.white, fontSize: 20, fontWeight: FontWeight.bold),
            ),
          ],
        ),
      ),
    );
  }
}

Task 4: Combined Row, Column & Stack Demo


import 'package:flutter/material.dart';

void main() {
  runApp(MyApp());
}

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Row, Column & Stack Demo',
      theme: ThemeData(primarySwatch: Colors.blue),
      home: HomeScreen(),
    );
  }
}

// Home Screen
class HomeScreen extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text("Flutter Layouts")),
      body: Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            ElevatedButton(
              child: Text("Row Layout"),
              onPressed: () {
                Navigator.push(
                  context,
                  MaterialPageRoute(builder: (context) => RowExample()),
                );
              },
            ),
            SizedBox(height: 20),
            ElevatedButton(
              child: Text("Column Layout"),
              onPressed: () {
                Navigator.push(
                  context,
                  MaterialPageRoute(builder: (context) => ColumnExample()),
                );
              },
            ),
            SizedBox(height: 20),
            ElevatedButton(
              child: Text("Stack Layout"),
              onPressed: () {
                Navigator.push(
                  context,
                  MaterialPageRoute(builder: (context) => StackExample()),
                );
              },
            ),
          ],
        ),
      ),
    );
  }
}
